cmake_minimum_required(VERSION 3.2)

set(PATCH_VERSION "1" CACHE INTERNAL "Patch version")
set(PROJECT_VERSION 0.0.${PATCH_VERSION})

project(mapallocatorv2 VERSION ${PROJECT_VERSION})

configure_file(
        versiondef.h.in
        ${CMAKE_CURRENT_BINARY_DIR}/versiondef.h
)

add_executable(mapallocatorv2 main.cpp)
add_library(allocator_lib version.cpp)

set_target_properties(mapallocatorv2 allocator_lib PROPERTIES
        CXX_STANDARD 17
        CXX_STANDARD_REQUIRED ON
        )

target_include_directories(
        allocator_lib
        PRIVATE
        ${CMAKE_CURRENT_BINARY_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
)

target_link_libraries(mapallocatorv2 PRIVATE
        allocator_lib
        )

if (MSVC)
    set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
    target_compile_options(mapallocatorv2 PRIVATE
            /W4
            )
    target_compile_options(allocator_lib PRIVATE
            /W4
            )
else ()
    target_compile_options(mapallocatorv2 PRIVATE
            -Wall -Wextra -pedantic -Werror
            )
    target_compile_options(allocator_lib PRIVATE
            -Wall -Wextra -pedantic -Werror
            )
endif()

install(TARGETS mapallocatorv2 RUNTIME DESTINATION bin)

set(CPACK_GENERATOR DEB)

set(CPACK_PACKAGE_VERSION_MAJOR "${PROJECT_VERSION_MAJOR}")
set(CPACK_PACKAGE_VERSION_MINOR "${PROJECT_VERSION_MINOR}")
set(CPACK_PACKAGE_VERSION_PATCH "${PROJECT_VERSION_PATCH}")

set(CPACK_PACKAGE_CONTACT zholu008@gmail.com)

include(CPack)
